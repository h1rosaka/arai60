```
class Solution(object):
    def hasCycle(self, head):
        """
        :type head: ListNode
        :rtype: bool
        """
        flag = False
        checked_nodes = []
        checking_node = head
        
        while checking_node.next != None:
            if checking_node in checked_nodes:
                flag = True
                break
            else:
                checked_nodes.append(checking_node)
                checking_node = checking_node.next

        return flag
```
