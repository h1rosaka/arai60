```python
class Solution:
    def deleteDuplicates(self, head: Optional[ListNode]) -> Optional[ListNode]:
        node = head

        while node is not None and node.next is not None:
            current_value = node.val

            if node.next.val == current_value:
                node.next = node.next.next
            else:
                node = node.next
                    
        return head
```
